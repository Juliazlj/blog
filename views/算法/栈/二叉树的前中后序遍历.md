## 前序
```javascript
var preorderTraversal = function(root) {
    const ret = [];
    const stk = [];
    let node = root;
    while(stk.length || node){
        while(node){
            ret.push(node.val);
            stk.push(node);
            node = node.left;
        }
        node = stk.pop();
        node = node.right;
    }
    return ret
};
```

## 中序
```javascript
var inorderTraversal = function(root) {
    const res = [];
    const stk = [];
    while (root || stk.length) {
        while (root) {
            stk.push(root);
            root = root.left;
        }
        root = stk.pop();
        res.push(root.val);
        root = root.right;
    }
    return res;
};
```
