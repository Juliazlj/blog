# Vue的全局方法

之前我们有看到在core/index里面调用了initGlobalAPI方法初始化Vue的全局方法，那么这一节我们来看看Vue的全局方法的实现

# set

我们知道如果给一个对象设置不存在的属性必须使用set方法，给数组下标设置属性也需要使用set方法，因为数组元素是不能被defineProperty劫持的。

直接点击initGlobalAPI里set的赋值可以找到set方法是在src/core/observer/index.js里定义的

export function set (target: Array<any> | Object, key: any, val: any): any {
  ...
  // 如果是数组就使用splice方法 因为vue是重写了splice，push，shift...等方法的 所以使用splice是可以添加响应的
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.length = Math.max(target.length, key)
    target.splice(key, 1, val)
    return val
  }
  // 若key已经存在了则直接设置
  if (key in target && !(key in Object.prototype)) {
    target[key] = val
    return val
  }
  // 获取target的观测对象属性
  const ob = (target: any).__ob__
	// 是Vue实例 或 根数据data时 报错
  if (target._isVue || (ob && ob.vmCount)) {
    process.env.NODE_ENV !== 'production' && warn(
      'Avoid adding reactive properties to a Vue instance or its root $data ' +
      'at runtime - declare it upfront in the data option.'
    )
    return val
  }
  // 若target不是观测对象 则表示target没在data里定义 不需要观测 则直接赋值
  if (!ob) {
    target[key] = val
    return val
  }
  // 将属性定义成响应式的
  defineReactive(ob.value, key, val)
  // 通知所有的依赖刷新视图
  ob.dep.notify()
  return val
}

# delete

原理同上，不赘述

export function del (target: Array<any> | Object, key: any) {
  ...
  if (Array.isArray(target) && isValidArrayIndex(key)) {
    target.splice(key, 1)
    return
  }
  const ob = (target: any).__ob__
  if (target._isVue || (ob && ob.vmCount)) {
    process.env.NODE_ENV !== 'production' && warn(
      'Avoid deleting properties on a Vue instance or its root $data ' +
      '- just set it to null.'
    )
    return
  }
  if (!hasOwn(target, key)) {
    return
  }
  delete target[key]
  if (!ob) {
    return
  }
  ob.dep.notify()
}

# nextTick

nextTick方法用处是让其回调函数在页面渲染完成后执行，这时我们已经可以获取到页面渲染完成后的dom元素。nextTick方法单独成为一个文件src/core/util/next-tick.js说明其重要性

export let isUsingMicroTask = false

const callbacks = []
let pending = false

// 清空nextTick队列
function flushCallbacks () {
  pending = false
  const copies = callbacks.slice(0)
  callbacks.length = 0
  for (let i = 0; i < copies.length; i++) {
    copies[i]()
  }
}

let timerFunc

// The nextTick behavior leverages the microtask queue, which can be accessed
// via either native Promise.then or MutationObserver.
// MutationObserver has wider support, however it is seriously bugged in
// UIWebView in iOS >= 9.3.3 when triggered in touch event handlers. It
// completely stops working after triggering a few times... so, if native
// Promise is available, we will use it:
/* istanbul ignore next, $flow-disable-line */

// Promise.then和MutationObserver都是微任务，但是MutationObserver在iOS >= 9.3.3有bug所以优先使用Promise.then
if (typeof Promise !== 'undefined' && isNative(Promise)) {
  const p = Promise.resolve()
  timerFunc = () => {
    p.then(flushCallbacks)
    // In problematic UIWebViews, Promise.then doesn't completely break, but
    // it can get stuck in a weird state where callbacks are pushed into the
    // microtask queue but the queue isn't being flushed, until the browser
    // needs to do some other work, e.g. handle a timer. Therefore we can
    // "force" the microtask queue to be flushed by adding an empty timer.
    
    // 有时候callbacks队列并不会清空 需要再加一个宏任务执行才能清空
    if (isIOS) setTimeout(noop)
  }
  isUsingMicroTask = true
  
  // 如果不支持Promise 则判断不是IE环境而且支持MutationObserver
} else if (!isIE && typeof MutationObserver !== 'undefined' && (
  isNative(MutationObserver) ||
  // PhantomJS and iOS 7.x
  MutationObserver.toString() === '[object MutationObserverConstructor]'
)) {
  // Use MutationObserver where native Promise is not available,
  // e.g. PhantomJS, iOS7, Android 4.4
  // (#6466 MutationObserver is unreliable in IE11)
  let counter = 1
  const observer = new MutationObserver(flushCallbacks)
  const textNode = document.createTextNode(String(counter))
  observer.observe(textNode, {
    characterData: true
  })
  timerFunc = () => {
    counter = (counter + 1) % 2
    textNode.data = String(counter)
  }
  isUsingMicroTask = true
  
  // 否则就使用setImmediate 这是个宏任务
} else if (typeof setImmediate !== 'undefined' && isNative(setImmediate)) {
  // Fallback to setImmediate.
  // Technically it leverages the (macro) task queue,
  // but it is still a better choice than setTimeout.
  timerFunc = () => {
    setImmediate(flushCallbacks)
  }
  // 再不济就使用setTImeout 这是个宏任务
} else {
  // Fallback to setTimeout.
  timerFunc = () => {
    setTimeout(flushCallbacks, 0)
  }
}

export function nextTick (cb?: Function, ctx?: Object) {
  let _resolve
  callbacks.push(() => {
    if (cb) {
      try {
        cb.call(ctx)
      } catch (e) {
        handleError(e, ctx, 'nextTick')
      }
    } else if (_resolve) {
      _resolve(ctx)
    }
  })
  // 多次调用nextTick只会执行一个timerFunc
  if (!pending) {
    pending = true
    timerFunc()
  }
  // 当用户没有传入cb的时候 就传回一个Promise 所以nextTick是支持Promise的 可以使用nextTick().then(...)
  if (!cb && typeof Promise !== 'undefined') {
    return new Promise(resolve => {
      _resolve = resolve
    })
  }
}

# use

use方法可以使用插件

Vue.use = function (plugin: Function | Object) {
  const installedPlugins = (this._installedPlugins || (this._installedPlugins = []))
  // 如果插件存在就直接返回
  if (installedPlugins.indexOf(plugin) > -1) {
    return this
  }

  // additional parameters
  // 把参数变成数组的格式 例如 Vue.use(Axios,1,2,3) => [1,2,3]
  const args = toArray(arguments, 1)
  // 在参数前面插入当前Vue的实例 [Vue,1,2,3] 为了让插件使用当前环境的Vue版本 不需要让插件强依赖某个版本的Vue
  args.unshift(this)
  // 如果有install方法则执行install
  if (typeof plugin.install === 'function') {
    plugin.install.apply(plugin, args)
  // 否则若plugin是函数 就执行该函数
  } else if (typeof plugin === 'function') {
    plugin.apply(null, args)
  }
  // 把插件存入缓存
  installedPlugins.push(plugin)
  return this
}
